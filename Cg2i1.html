   파일 열기 
   원본 파일 끝까지 한 줄씩 읽기

   파일 닫기 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter12/FileCopy.c

	#include <stdio.h>

	 

	// 프로그램시작 

1	int main() {

		char line[1024] = "";

2		FILE *inFile = NULL, *outFile = NULL;

 

		// 파일 열기 

3		if (((inFile=fopen("original.txt","rt"))==NULL )||((outFile=fopen("copy.txt","wt"))==NULL)) {

			printf( "[오류] 파일을 열 수 없습니다!" );

4			return -1;

		}

 

		// 원본 파일 끝까지 한 줄씩 읽기 

5		while ( fgets( line, 1024, inFile ) != NULL ) {

			// 복사본 파일에 한 줄씩 쓰기 

6			fprintf( outFile, "%s", line );

		}

 

		// 파일 닫기 

		fclose( inFile );

7		fclose( outFile );

	 

		// 프로그램 종료 

8		return 0;

	}
